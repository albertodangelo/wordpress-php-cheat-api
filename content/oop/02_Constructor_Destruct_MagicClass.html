class User {

    // Properties / Eigenchaften
    public $name = "";
    public $age = "";

    public function __construct($name, $age) {
        echo "Klasse " . __CLASS__ . ' gestartet
';
        
        $this->name = $name;
        $this->age = $age;
    }
    
    // Methods / Funktionen
    public function sayHello() {
        return $this->name . ' sagt Hallo.';
    }
    
    public function __destruct() {
        echo "
Klasse " . __CLASS__ . ' wurde beendet
';

    }
}

$user1 = new User('Alberto', 30);
echo $user1->name . " ist " . $user1->age . ' Jahre alt';
echo '
';
echo $user1->sayHello();
echo '
';
echo '
';
echo '
